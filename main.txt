1.
RETRIEVE NAMES AND PHONE NUMBERS OF ALL FEMALE STUDENTS
SELECT SName, Phone
FROM STUDENT
WHERE Gender = 'Female';

2.
LIST STUDENT NAMES ALONG WITH THEIR SEMESTER AND SECTION
SELECT SName, Sem, Sec
FROM STUDENT, CLASS, SEMSEC
WHERE STUDENT.USN = CLASS.USN
AND CLASS.SSID = SEMSEC.SSID;

3.
LIST ALL STUDENTS WITH THEIR INTERNAL MARKS IN EACH SUBJECT
SELECT SName, Subcode, Test1, Test2, Test3, FinalIA
FROM STUDENT, IA_MARKS
WHERE STUDENT.USN = IA_MARKS.USN;

4.
TOTAL NUMBER OF MALE AND FEMALE STUDENTS IN EACH SEMESTER AND SECTION
SELECT Sem, Sec, Gender, COUNT(*) AS Total
FROM STUDENT, CLASS, SEMSEC
WHERE STUDENT.USN = CLASS.USN
AND CLASS.SSID = SEMSEC.SSID
GROUP BY Sem, Sec, Gender;

5.
RETRIEVE STUDENTS WHO SCORED MORE THAN 20 IN ALL THREE TESTS
SELECT USN
FROM IA_MARKS
WHERE Test1 > 20 AND Test2 > 20 AND Test3 > 20;

6.
CALCULATE FINALIA (AVERAGE OF BEST TWO TEST MARKS) AND UPDATE TABLE
UPDATE IA_MARKS
SET FinalIA = (
(Test1 + Test2 + Test3
- LEAST(Test1, Test2, Test3)) / 2
);

7.
CATEGORIZE STUDENTS (ONLY 8TH SEM A, B, C SECTIONS)
SELECT SName, Sem, Sec, FinalIA,
CASE
WHEN FinalIA BETWEEN 21 AND 25 THEN 'Outstanding'
WHEN FinalIA BETWEEN 16 AND 20 THEN 'Average'
ELSE 'Weak'
END AS Category
FROM STUDENT, CLASS, SEMSEC, IA_MARKS
WHERE STUDENT.USN = CLASS.USN
AND CLASS.SSID = SEMSEC.SSID
AND STUDENT.USN = IA_MARKS.USN
AND Sem = 8
AND Sec IN ('A', 'B', 'C');

1.
FIND NAMES OF ACTORS WHO ACTED IN MOVIES RELEASED IN THE YEAR 2022
SELECT Act_Name
FROM ACTOR, MOVIES, MOVIE_CAST
WHERE ACTOR.Act_id = MOVIE_CAST.Act_id
AND MOVIES.Mov_id = MOVIE_CAST.Mov_id
AND Mov_Year = 2022;

2.
FIND THE TITLES OF MOVIES THAT HAVE THE HIGHEST RATING
SELECT Mov_Title
FROM MOVIES, RATING
WHERE MOVIES.Mov_id = RATING.Mov_id
AND Rev_Stars = (
SELECT MAX(Rev_Stars) FROM RATING
);

3.
LIST DIRECTORS WHO HAVE DIRECTED MOVIES IN MORE THAN ONE
LANGUAGE
SELECT Dir_Name
FROM DIRECTOR, MOVIES
WHERE DIRECTOR.Dir_id = MOVIES.Dir_id
GROUP BY Dir_Name
HAVING COUNT(DISTINCT Mov_Lang) > 1;

4.
FIND ALL MOVIE TITLES THAT ARE EITHER IN ENGLISH OR HAVE A RATING OF 5
STARS
SELECT Mov_Title
FROM MOVIES, RATING
WHERE MOVIES.Mov_id = RATING.Mov_id
AND (Mov_Lang = 'English' OR Rev_Stars = 5);

5.
FIND COMMON ACTORS WHO ACTED IN BOTH 2020 AND 2023 RELEASED MOVIES
SELECT Act_Name
FROM ACTOR, MOVIE_CAST MC1, MOVIES M1, MOVIE_CAST MC2, MOVIES
M2
WHERE ACTOR.Act_id = MC1.Act_id
AND MC1.Mov_id = M1.Mov_id
AND ACTOR.Act_id = MC2.Act_id
AND MC2.Mov_id = M2.Mov_id
AND M1.Mov_Year = 2020
AND M2.Mov_Year = 2023;

6.
CREATE A VIEW THAT LISTS MOVIE TITLE, DIRECTOR NAME, AND RATING
CREATE VIEW MOVIE_DETAILS AS
SELECT Mov_Title, Dir_Name, Rev_Stars
FROM MOVIES, DIRECTOR, RATING
WHERE MOVIES.Dir_id = DIRECTOR.Dir_id
AND MOVIES.Mov_id = RATING.Mov_id
