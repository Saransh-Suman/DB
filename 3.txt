CREATE TABLE ACTOR (
    ACT_ID NUMBER PRIMARY KEY,
    ACT_NAME VARCHAR2(100) NOT NULL,
    ACT_GENDER VARCHAR2(10)
);

CREATE TABLE DIRECTOR (
    DIR_ID NUMBER PRIMARY KEY,
    DIR_NAME VARCHAR2(100) NOT NULL,
    DIR_PHONE VARCHAR2(15)
);

CREATE TABLE MOVIES (
    MOV_ID NUMBER PRIMARY KEY,
    MOV_TITLE VARCHAR2(150) NOT NULL,
    MOV_YEAR NUMBER,
    MOV_LANG VARCHAR2(50),
    DIR_ID NUMBER,
    FOREIGN KEY (DIR_ID) REFERENCES DIRECTOR(DIR_ID)
);

CREATE TABLE MOVIE_CAST (
    ACT_ID NUMBER,
    MOV_ID NUMBER,
    ROLE VARCHAR2(100),
    PRIMARY KEY (ACT_ID, MOV_ID),
    FOREIGN KEY (ACT_ID) REFERENCES ACTOR(ACT_ID),
    FOREIGN KEY (MOV_ID) REFERENCES MOVIES(MOV_ID)
);

CREATE TABLE RATING (
    MOV_ID NUMBER PRIMARY KEY,
    REV_STARS NUMBER,
    FOREIGN KEY (MOV_ID) REFERENCES MOVIES(MOV_ID)
);

-- 1. Find names of actors who acted in movies released in the year 2022.
-- 2. Find the titles of movies that have the highest rating.
-- 3. List directors who have directed movies in more than one language.
-- 4. Find all movie titles that are either in English or have a rating of 5 stars.
-- 5. Find common actors who acted in both 2020 and 2023 released movies.
-- 6. Create a VIEW that lists movie title, director name, and rating.

SELECT ACT_NAME
FROM ACTOR, MOVIE_CAST, MOVIES
WHERE ACTOR.ACT_ID = MOVIE_CAST.ACT_ID
AND MOVIE_CAST.MOV_ID = MOVIES.MOV_ID

SELECT MOV_TITLE
FROM MOVIES, RATING
WHERE MOVIES.MOV_ID = RATING.MOV_ID
AND RATING.REV_STARS = (SELECT MAX(REV_STARS) FROM RATING);

SELECT DIR_NAME FROM DIRECTOR, MOVIES
WHERE DIRECTOR.DIR_ID = MOVIES.DIR_ID
GROUP BY DIR_NAME
HAVING COUNT(DISTINCT MOV_LANG)>1;

SELECT Mov_Title
FROM MOVIES, RATING
WHERE MOVIES.Mov_id = RATING.Mov_id
AND (Mov_Lang = 'English' OR Rev_Stars = 5);

SELECT Act_Name
FROM ACTOR, MOVIE_CAST MC1, MOVIES M1, MOVIE_CAST MC2, MOVIES
M2
WHERE ACTOR.Act_id = MC1.Act_id
AND MC1.Mov_id = M1.Mov_id
AND ACTOR.Act_id = MC2.Act_id
AND MC2.Mov_id = M2.Mov_id
AND M1.Mov_Year = 2020
AND M2.Mov_Year = 2023;

CREATE VIEW MOVIE_DETAILS AS
SELECT Mov_Title, Dir_Name, Rev_Stars
FROM MOVIES, DIRECTOR, RATING
WHERE MOVIES.Dir_id = DIRECTOR.Dir_id
AND MOVIES.Mov_id = RATING.Mov_id;
