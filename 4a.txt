CREATE TABLE CUSTOMERS (
ID NUMBER,
NAME VARCHAR2(20),
AGE NUMBER,
ADDRESS VARCHAR2(50),
SALARY NUMBER
);
INSERT INTO CUSTOMERS VALUES (1,'Ravi',30,'Bangalore',40000);
INSERT INTO CUSTOMERS VALUES (2,'Meena',28,'Mysore',35000);
INSERT INTO CUSTOMERS VALUES (3,'Arun',35,'Hubli',45000);

CREATE OR REPLACE TRIGGER salary_difference
BEFORE INSERT OR UPDATE OR DELETE
ON CUSTOMERS
FOR EACH ROW
BEGIN
IF INSERTING THEN
DBMS_OUTPUT.PUT_LINE('Inserted Salary: ' || :NEW.SALARY);
ELSIF UPDATING THEN
DBMS_OUTPUT.PUT_LINE('Old Salary: ' || :OLD.SALARY);
DBMS_OUTPUT.PUT_LINE('New Salary: ' || :NEW.SALARY);
DBMS_OUTPUT.PUT_LINE('Salary Difference: ' || (:NEW.SALARY -
:OLD.SALARY));
ELSIF DELETING THEN
DBMS_OUTPUT.PUT_LINE('Deleted Salary: ' || :OLD.SALARY);
END IF;
END;
/
SET SERVEROUTPUT ON;
UPDATE CUSTOMERS
SET SALARY = 50000
WHERE ID = 1;
